Klasse Signatur

Attribute:
    privat PPN
    privat ExemplarNr
    privat SignaturenText
    privat BearbeitetAm

Methoden:
    setzePPN(str){
        this.PPN = str
    }

    setzeExemplarNr(str){
        this.ExemplarNr = str
    }

    setzeSignaturenText(str){
        this.SignaturenText = str
    }

    setzeBearbeitetAm(str){
        this.BearbeitetAm = str
    }

    getSignatur(){
        return {[
            'PPN': this.PPN,
            'ExemplarNr': this.ExemplarNr,
            'SignaturenText': this.SignaturenText,
            'BearbeitetAm': this.BearbeitetAm]}
    }

    checkAllSet(){
        if (this.PPN && this.ExemplarNr && SignaturenText && BearbeitetAm){
            return true
        } else {
            return false
        }
    }
